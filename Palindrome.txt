// Program to find palindrome number
#include <stdio.h>

int main() 
{
    // Write C code here
    int n, rem, rev=0, temp;
    
    printf("Enter a Number: ");
    scanf("%d", &n);
   temp=n;
    while(n>0)
    {
        rem=n%10;
        rev=rev*10+rem;
        n=n/10;
    }
   n=temp;
   
    if(n==rev)
    printf("Number is Palindrome", n);
    else
    printf("Number is not a Palindrome", n);
    return 0;
}
